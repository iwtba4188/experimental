    /* Render */
    function renderSchedule(rootId, weekdayBusSchedule, weekendBusSchedule) {
      // 呈現 平日 or 假日 的時刻資訊
      let busSchedule = weekdayBusSchedule;
      if (isWeekend()) {
        busSchedule = weekendBusSchedule
      }

      const root = document.getElementById(rootId);
      root.innerHTML = '';
    
      const scheduleContainer = document.createElement("div");
      scheduleContainer.setAttribute("class", "horizone-container");

      /* 標頭 */
      const scheduleHeader = document.createElement("div");
      scheduleHeader.setAttribute("class", "horizone-row schedule-header");

      const indexHeaderNode = document.createElement('div')
      indexHeaderNode.setAttribute("class", "horizone-col-xs-3");
      indexHeaderNode.textContent = '編號 (#)';

      const timeHeaderNode = document.createElement('div')
      timeHeaderNode.setAttribute("class", "horizone-col-xs-3");
      timeHeaderNode.textContent = '時刻 (Dep. Time)';

      const descriptionHeaderNode = document.createElement('div')
      descriptionHeaderNode.setAttribute("class", "horizone-col-xs-6");
      descriptionHeaderNode.textContent = '備註 (Note)';

      scheduleHeader.append(indexHeaderNode, timeHeaderNode, descriptionHeaderNode);
      scheduleContainer.append(scheduleHeader);

      /* 時刻內容 */
      for (let index = 0; index < busSchedule.length; index++) {
        const scheduleRow = document.createElement("div");
        scheduleRow.setAttribute("class", "horizone-row schedule-row");

        const item = busSchedule[index];

        // 判斷過時的時刻
        const busHourMinute = item.time.split(':');
        if (isTimeAfter(busHourMinute[0], busHourMinute[1])) {
          scheduleRow.classList.add("schedule-row--disabled");
        }
        
        // 編號
        const indexNode = document.createElement('div')
        indexNode.setAttribute("class", "horizone-col-xs-3");
        indexNode.textContent = `#${index + 1}`;

        // 時刻
        const timeNode = document.createElement('div');
        timeNode.setAttribute("class", "horizone-col-xs-3 row-item item--focus");
        timeNode.textContent = item.time;

        // 說明
        const descriptionNode = document.createElement('div');
        descriptionNode.setAttribute("class", "horizone-col-xs-6 row-item item--description");
        descriptionNode.textContent = item.description;

        // 放入 Row
        scheduleRow.append(indexNode, timeNode, descriptionNode);
        scheduleContainer.append(scheduleRow)

        // 放入 Divider
        if (index < busSchedule.length - 1) {
          const dividerNode = document.createElement('div');
          dividerNode.setAttribute("class", "schedule-row--divider");
          scheduleContainer.append(dividerNode)
        }
      }

      root.append(scheduleContainer);
    }

    /* Utils */
    function isWeekend() {
      const today = new Date();
      const dayOfWeek = today.getDay();
      return dayOfWeek === 0 || dayOfWeek === 6;
    }

    function isTimeAfter(targetHour, targetMinute) {
      const now = new Date();
      
      const currentHour = now.getHours();
      const currentMinute = now.getMinutes();
      
      const targetTimeInMinutes = parseInt(targetHour) * 60 + parseInt(targetMinute);
      const currentTimeInMinutes = currentHour * 60 + currentMinute;
      
      return currentTimeInMinutes > targetTimeInMinutes;
    }

function openLangUrl(p_lang) {
   var curlang = getUrlParam("Lang");
   if(curlang!="") {
      var str = window.location+"";
      window.location = str.replace("Lang="+curlang,"Lang="+p_lang);
   }
   else {
      if(window.location.href.indexOf("?")>0)
         window.location=window.location+'&Lang='+p_lang;
      else
         window.location=window.location+'?Lang='+p_lang;
   }
}
function getUrlParam(paramName,url){
	if(typeof url=="undefined") url=window.location;
   var oRegex = new RegExp( '[\?&]' + paramName + '=([^&]+)', 'i' ) ;
   var oMatch = oRegex.exec(url) ;
   if ( oMatch && oMatch.length > 1 )
      return oMatch[1] ;
   else
      return '' ;
}
